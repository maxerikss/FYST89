\section{Scattering Theory}
Scattering processes are the processes that transform an initial state into a final state, by some potential, which we will treat as a time-dependent perturbation. This is how we learn about experimental processes.

\subsection{Scattering as a Time-Dependent Perturbation}
We make the assumption that the Hamiltonian may be written as
\begin{equation}
    H = H_0 + V(\vec{x}),
\end{equation}
with
\begin{equation}
    H_0 = \frac{\vec{p}^2}{2m},
\end{equation}
the kinetic energy operator. The eigenvalues of $H_0$ are 
\begin{equation}
    E_{\vec{k}} = \frac{\hbar^2 \vec{k}^2}{2m}.
\end{equation}
Plane wave eigenvectors of $H_0$, with eigenvalues $E_{\vec{k}}$, are $\k{\vec{k}}$, and we make the assumption that the scattering potential $V(\vec{r})$ is time independent. We will treat this as an incoming particle seeing the scattering potential which is active only when the particle is close to the scatterer. This analysis will be performed using time-dependent perturbation theory in the interaction picture.

We know that the state evolution is
\begin{equation}
    \k{\alpha, t_0; t}_I = U_I(t,t_0)\k{\alpha, t_0;t_0},
\end{equation}
where $U_I(t,t_0)$ is the time evolution operator in the interaction picture, satisfying
\begin{equation}
    i\hbar \partial_t U_I(t,t_0) = V_I(t)U_I(t,t_0).
\end{equation}
We have $U_I(t_0,t_0) = 1$ and $V_I(t)=\exp(iH_0t/\hbar) V \exp(-iH_0t/\hbar)$, and the exact solution to the equation may be written as
\begin{equation}
    U_I(t,t_0) = 1 - \frac{i}{\hbar}\int_{t_0}^t dt' V_I(t')U_I(t',t_0).
\end{equation}
The transition amplitude for the transformation from an initial state $\k{i}$ into a final state $\k{n}$, both eigenstate to $H_0$, can be written as 
\begin{equation}
    \bkop{n}{U_I(t,t_0)}{i} = \delta_{ni} - \frac{i}{\hbar} \sum_m \bkop{n}{V}{m} \int_{t_0}^{t} dt' e^{i \omega_{nm} t'} \bkop{m}{U_I(t',t_0)}{i},
\end{equation}
where $\hbar\omega_{nm} = E_n - E_m$ and $\delta{ni}=\bk{n}{i}$. We arrive here by inserting the completeness relation $\sum_m \kb{m}{m}$, and converting $V_I \to V$. 

In scattering theory we deal with a continuum of states, not discrete ones, we therefore have to adjust the above equation to account for this. We do this by quantizing the scattering states in a cubic volume of side $L$. We have, in the coordinate representation, the wave function
\begin{equation}
    \bk{\vec{x}}{\vec{k}} = \frac{1}{L^{3/2}} e^{i\vec{k}\cdot\vec{x}},
\end{equation}
where we will take $L\to\infty$ at the end. Note that $\bk{\vec{k'}}{\vec{k}} = \delta_{\vec{k'}\vec{k}}$.

The initial and final states exist only asymptomatically, and we therefore have to have $t\to\infty$ and $t_0 \to -\infty$.

We start by doing a first order treatment, where we set $\bkop{m}{U_I(t',t_0)}{i} = \delta_{mi}$, which gives
\begin{equation}
    \bkop{n}{U_I(t,t_0)}{i} = \delta_{ni} - \frac{i}{\hbar} \bkop{n}{V}{i} \int_{t_0}^{t} dt' e^{i\omega_{ni}t'}.
\end{equation}
For the $t\to\infty$ we have Fermi's golden rule, and in order to include the behaviour for $t_0 \to - \infty$ we define a matrix $T_{ni}$ by
\begin{equation}
    \bkop{n}{U_I{t,t_0}}{i} = \delta_{ni} - \frac{i}{\hbar} T_{ni} \int_{t_0}^{t} dt' e^{i\omega_{ni}t' + \epsilon t'}
\end{equation}
with $\epsilon > 0$ and $t \ll 1/\epsilon$, and we need to make sure to take the limit $\epsilon \to 0$ before we take $t \to \infty$. We now define the scattering matrix
\begin{equation}
    S_{ni} = \lim_{t\to \infty} \ab[\lim_{\epsilon\to 0} \bkop{n}{U_I{t,-\infty}}{i}] = \delta_{ni} - \frac{i}{\hbar} T_{ni} \int_{-\infty}^{\infty} dt' e^{i\omega{ni} t'} = \delta_{ni} - 2\pi i \delta(E_n - E_i) T_{ni}.
\end{equation}
We can see that the scattering matrix consists of two parts. One part where the final and initial states are the same, and one part governed by the $T$-matrix, with some sort of scattering.

\subsubsection{Transition Rates and Cross Sections}
We define that transition rate as
\begin{equation}
    w(i\to n) = \partial_t  \abs{\bkop{n}{U_I{t,-\infty}}{i}}^2
\end{equation}
and for $\k{i} \neq \k{n}$, we have
\begin{equation}
    \bkop{n}{U_I(t,-\infty)}{i} = - \frac{i}{\hbar} T_{ni} \int_{-\infty}^{t} dt' e^{i\omega_{ni} t' + \epsilon t'} = -\frac{i}{\hbar} T_{ni} \frac{e^{i\omega_{ni} t + \epsilon t}}{i \omega_{ni} + \epsilon}.
\end{equation}
The transition rate for this is then
\begin{equation}
    w(i \to n) = \partial_t \ab [\frac{1}{\hbar^2} \abs{T_{ni}}^2 \frac{e^{2\epsilon t}}{\omega_{ni}^2 + \epsilon^2} ] = \frac{1}{\hbar^2} \abs{T_{ni}}^2 \frac{2\epsilon e^{2\epsilon t}}{\omega_{ni}^2 + \epsilon^2}.
\end{equation}
Here, we need to take $\epsilon \to 0$ first, before taking $t \to \infty$. If $\omega_{ni}\neq 0$, we see that $w \to 0$, while for $\omega_{ni} = 0$, we get a delta function. We find that 
\begin{equation}
    \frac{e e^{2\epsilon t}}{\omega_{ni}^2 + \epsilon^2} = \pi \delta(\omega_{ni}) = \pi \hbar \delta(E_n - E_i),
\end{equation}
and we get the transmission rate
\begin{equation}
    w(i \to n) = \frac{2\pi}{\hbar} \abs{T_{ni}}^2 \delta(E_n - E_i)
\end{equation}
which is independent of time so the limit $t\to\infty$ is trivial. This is very similar to Fermi's golden rule, but with $T_{ni}$ instead of $V_{ni}$, which is a more general matrix. We can use this transition rate to determine a scattering cross section.

Similarly to Fermi's golden rule, we need to determine the density of final states, $\rho(E_n) = \Delta n /\Delta E_n$, so that we can integrate over the final state energy. We will consider elastic scattering, where $\k{i} = \k{\vec{k}}, \k{n}=\k{\vec{k}'}$, and $\abs{\vec{k}} = \abs{\vec{k}'}= k$. Using the normalization due to our cubic volume, we can write
\begin{equation}
    E_{\vec{n}} = \frac{\hbar^2 \vec{k}'^2}{2 m} = \frac{\hbar^2}{2m} \ab (\frac{2\pi}{L})^2 \abs{\vec{n}}^2,
\end{equation}
and thus
\begin{equation}
    \Delta E_{\vec{n}} = \partial_{\vec{n}} (\Delta E_{\vec{n}}) \Delta\abs{\vec{n}} = \frac{\hbar^2}{m} \ab(\frac{2\pi}{L})^2 \abs{\vec{n}} \Delta \abs{\vec{n}},
\end{equation}
where $\vec{n} = n_x \hv{i} + n_y \hv{j} + n_z \hv{k}$ and $n_{x,y,z} \in \mathbb{Z}$. We have $\vec{n} = (L/2\pi) \abs{\vec{k}'} = (L/2\pi)k$ and $L$ is large, so we can imagine $\abs{\vec{n}}$ as almost continuous. The number of states within a spherical shell of radius $\abs{\vec{n}}$ and thickness $\Delta\abs{\vec{n}}$ is 
\begin{equation}
    \Delta n = 4\pi \abs{\vec{n}}^2 \Delta \abs{\vec{n}} \times \frac{d\Omega}{4\pi},
\end{equation}
considering the fraction of solid angle represented by the final state vector $\vec{k}'$. Thus,
\begin{equation}
    \rho(E_n) = \frac{\Delta n}{\Delta E_n} = \frac{m}{\hbar^2} \ab(\frac{L}{2\pi})^2 \abs{\vec{n}} d\Omega = \frac{mk}{\hbar^2} \ab(\frac{L}{2\pi})^3 d\Omega.
\end{equation}
Integrating over the final states gives
\begin{equation}
    w(i\to n) = \frac{m k L^3}{(2\pi)^2 \hbar^3} \abs{T_{ni}}^2 d\Omega.
\end{equation}

This concept of cross section is used to understand the transition rate in scattering experiments. Determining the rate at which particles scatter into a solid angle $d\Omega$ from a beam of particles with momentum $\hbar\vec{k}$. The speed of the particles is $v = \hbar k /m$, and the time needed to cross the cubic volume is $t = L/v$. The particle flux in the beam is $v / L^3$, and the probability flux for the wave function becomes
\begin{equation}
    \vec{j}(\vec{x}, t) = \ab (\frac{\hbar}{m}) \frac{\vec{k}}{L^3} = \frac{v}{L^3}.
\end{equation}
The cross section $d\sigma$ is defined as the transition rate divided by the flux. Finally, we arrive at
\begin{equation}
    \frac{d\sigma}{d\Omega} = \ab(\frac{mL^3}{2 \pi\hbar^2})^2 \abs{T_{ni}}^2.
\end{equation}
We now have to relate the matrix elements $T_{ni}$ to the scattering potential $V(\vec{r})$.

\subsubsection{Solving for the $T$ matrix}
Recalling the definition of the $T$ matrix we have
\begin{equation}
    \bkop{n}{U_I(t,-\infty)}{i} = \delta_{ni} + \frac{1}{\hbar} T_{ni} \frac{e^{i\omega_{ni}t + \epsilon t}}{-\omega_{ni}+ i\epsilon}
\end{equation}
and 
\begin{equation}
    \bkop{n}{U_I(t,-\infty)}{i} = \delta_{ni} - \frac{i}{\hbar} \sum_m V_{nm} \int_{-\infty}^{t} dt' e^{i\omega_{nm}t'} \bkop{m}{U_I(t', -\infty)}{i}.
\end{equation}
Inserting the former equation into the latter under the integral sign we obtain three terms. The first of which is $\delta_{ni}$, the second looks like the former equation but with $T_{ni}$ replaced by $V_{ni}.$ The final term is
\begin{equation}
    -\frac{i}{\hbar} \frac{1}{\hbar} \sum_m V_{nm} \frac{T_{mi}}{-\omega_{mi} + i\epsilon} \int_{-\infty}^{t} dt' e^{i(\omega_{nm} + \omega_{mi})t' + \epsilon t'}.
\end{equation}
Since $\omega_{nm} + \omega_{mi} = \omega_{ni}$, the result of the integral can be taken out from the summation. Gathering terms and comparing to the former equation we find that
\begin{equation}
    T_{ni} = V_{ni} + \frac{1}{\hbar} \sum_m V_{nm} \frac{T_{mi}}{-\omega_{mi} + i\epsilon} = V_{ni} + \sum_m V_{nm} \frac{T_{mi}}{E_i - E_m + i\hbar\epsilon}.
\end{equation}
This is an inhomogeneous system of linear equation which can be solved for $T_{ni}$ in terms of the matrix elements $V_{nm}$, which are known.

We can define a set of vectors $\k{\psi^{(+)}}$ in terms of components in some basis $\k{j}$, such that
\begin{equation}
    T_{ni} = \sum_j \bkop{n}{V}{j}\bk{j}{\psi^{(+)}} = \bkop{n}{V}{\psi^{(+)} }.
\end{equation}
Inserting this into the equation for $T_{ni}$, we find
\begin{equation}
    \bkop{n}{V}{\psi^{(+)} } = \bkop{n}{V}{i} + \sum_m \bkop{n}{V}{m} \frac{\bkop{m}{V}{\psi^{(+)}}}{E_i - E_m + i\hbar \epsilon}.
\end{equation}
Since this must be true for all $\k{n}$ we find 
\begin{equation}
    \k{\psi^{(+)}} = \k{i} + \frac{1}{E_i - H_0 + i\hbar \epsilon} V\k{\psi^{(+)}},
\end{equation}
which is known as the Lippmann-Schwinger equation. We can define an operator $T$ with matrix elements $T_{ni}$ by writing
\begin{equation}
    T\k{i} = V \k{\psi^{(+)}},
\end{equation}
we can then find the operator equation
\begin{equation}
    T = V + V\frac{1}{E_i - H_0 + i\hbar\epsilon}T.
\end{equation}

\subsubsection{Scattering from the Future to the Past}
We could instead imagine the scattering process evolving backwards in time, from a plane wave state $\k{i}$ in the far future, to a state $\k{n}$ in the distant past. We would in this case write 
\begin{equation}
    U_I(t,t_0) = 1 + \frac{i}{\hbar} \int_{t}^{t_0} dt' V_I(t') U_I(t',t_0)
\end{equation}
and the $T$ matrix is the n defined as 
\begin{equation}
    \bkop{n}{U_I(t,t_0)}{i} = \delta_{ni} + \frac{1}{\hbar} T_{ni} \int_{t}^{t_0} dt' e^{i\omega_{ni} t' - \epsilon t'}.
\end{equation}
In this case we can define $T$ using a different set of states $\k{\psi^{(-)}}$ through
\begin{equation}
    T\k{i} = V \k{\psi^{(-)}}.
\end{equation}

\subsection{The Scattering Amplitude}
We make the substitution $\hbar \epsilon \to \epsilon$ in the Lippmann-Schwinger equation. We continue considering elastic scattering and we use $E$ for the initial (and final) energy. We write
\begin{equation}
    \k{\psi^{(\pm)}} = \k{i} + \frac{1}{E-H_0 \pm i\epsilon} V\k{\psi^{(\pm)}}
\end{equation}
and multiply by $\b{\vec{x}}$ from the left, and by inserting the completeness relation for position states we get
\begin{equation}
    \bk{\vec{x}}{\psi^{(\pm)} } = \bk{\vec{x}}{i} + \int d^3 x' \bkop{\vec{x}}{\frac{1}{E - H_0 \pm i\epsilon}}{\vec{x}'}\bkop{\vec{x}'}{V}{\psi^{(\pm)} },
\end{equation}
which is an integral equation for scattering. The first step is to evaluate the function
\begin{equation}
    G_\pm (\vec{x},\vec{x}') = \frac{\hbar^2}{2m} \bkop{\vec{x}}{\frac{1}{E - H_0 \pm i \epsilon}}{\vec{x}'}.
\end{equation}
The eigenstates of $H_0$ are easier to evaluate in the momentum basis, and we therefore insert complete sets of these. Doing this, we obtain
\begin{equation}
    G_\pm (\vec{x},\vec{x}') = \frac{\hbar^2}{2m} \sum_{\vec{k}'}\sum_{\vec{k}''} \bk{\vec{x}}{\vec{k}'} \bkop{\vec{k}'}{\frac{1}{E-H_0\pm i\epsilon}}{\vec{k}''} \bk{\vec{k}''}{\vec{x}'}
\end{equation}
and letting $H_0$ act on $\k{\vec{k}'}$ we find
\begin{equation}
    \bkop{\vec{k}'}{\frac{1}{E-H_0\pm i\epsilon}}{\vec{k}''} =  \frac{\delta_{\vec{k}'\vec{k}''}}{E - (\hbar^2 \vec{k}'' / 2m) \pm i\epsilon}.
\end{equation}
Recalling
\begin{align}
    \bk{\vec{x}}{\vec{k}'} &= \frac{e^{i\vec{k}'\cdot \vec{x}}}{L^{3/2}}\\
    \bk{\vec{k}''}{\vec{x}'} &= \frac{e^{-i\vec{k}''\cdot \vec{x}'}}{L^{3/2}}
\end{align}
and writing $E = \hbar^2 k^2 / 2m$, we find 
\begin{equation}
    G_\pm (\vec{x}, \vec{x}') = \frac{1}{L^3} \sum_{\vec{k}'} \frac{e^{i\vec{k}'\cdot(\vec{x}-\vec{x}')}}{k^2 - k'^2 \pm i\epsilon}.
\end{equation}
We can convert this to an integral by using
\begin{equation}
    \sum_{\vec{k}} \to \frac{L^3}{(2\pi)^3} \int d^3 k,
\end{equation}
as $L \to \infty$. We now get
\begin{equation}
    G_\pm (\vec{x}, \vec{x}') = \frac{1}{(2\pi)^3} \int d^3 k' \frac{e^{i\vec{k}'\cdot(\vec{x} - \vec{x}')}}{k^2 - k'^2 \pm i \epsilon}.
\end{equation}
This is analytically solvable by switching to spherical coordinates and using contour integration and residuals. Using a contour in either the upper or lower half plane depending on where the singularity is, and applying Cauchy's integral formula, we find
\begin{equation}
    G_\pm (\vec{x}, \vec{x}') = -\frac{1}{4\pi} \frac{e^{\pm i  k\abs{\vec{x} - \vec{x}'}}}{\abs{\vec{x}-\vec{x}'}}
\end{equation}
We recognize that $G_\pm$ is the Green function for the Helmholtz equation
\begin{equation}
    (\nabla^2 + k^2)G_\pm(\vec{x},\vec{x}') = \delta^{(3)}(\vec{x}-\vec{x}'),
\end{equation}
and for $\vec{x}\neq \vec{x}'$ the eigenvalue euqation is solved by $G_\pm$
\begin{equation}
    H_0 G_\pm = E G_\pm.
\end{equation}
Inserting this into the equation for the wave function we find
\begin{equation}
    \bk{\vec{x}}{\psi^{(\pm)}} = \bk{\vec{x}}{i} - \frac{2m}{\hbar^2} \int d^3 x' \frac{e^{\pm ik\abs{\vec{x}-\vec{x}'}}}{4\pi\abs{\vec{x}-\vec{x}'}} \bkop{\vec{x}'}{V}{\psi^{(\pm)}}.
\end{equation}
Note thtat this consists of two terms. The first terms is the incoming wave, while the second is the effect of scattering. Thus, without any scattering, the scattered wave is the same as the incident wave. We can show that the positive solution $\psi^{(+)}$ corresponds to a plane wave plus and outgoing spherical wave, while the negative solution $\psi^{(-)}$ corresponds to a plane wave plus and incoming spherical wave. Which we see as scatterinf forwards or backwards in time. Usually we are interested in the positive solution in physics.

Consider a local potential $V$, that is, a potential only a function of the position operator $\vec{x}$, and diagonal in the position representation. We can write such a local potential as
\begin{equation}
    \bkop{\vec{x}'}{V}{\vec{x}''} = V(\vec{x}') \delta^{(3)}(\vec{x'}-\vec{x}'').
\end{equation}
By inserting a complete set into $\bkop{\vec{x}'}{V}{\psi^{(\pm)}}$ we get
\begin{equation}
    \bkop{\vec{x}'}{V}{\psi^{(\pm)}} = \int d^3 x'' \bkop{\vec{x}'}{V}{\vec{x}''}\bk{\vec{x}''}{\psi^{(\pm)}} = V(\vec{x}')\bk{\vec{x}'}{\psi^{(\pm)}},
\end{equation}
which we can use to simplify the equation for the wave function
\begin{equation}
    \bk{\vec{x}}{\psi^{(\pm)}} = \bk{\vec{x}}{\psi^{(\pm)}} = \bk{\vec{x}}{i} - \frac{2m}{\hbar^2} \int d^3 x' \frac{e^{\pm ik\abs{\vec{x}-\vec{x}'}}}{4\pi\abs{\vec{x}-\vec{x}'}} V(\vec{x}') \bk{\vec{x}'}{\psi^{(\pm)}}. 
\end{equation}
We understand $\vec{x}$ as the vector pointing to the observation point, where we measure, or evaluate, the wave function. The vector $\vec{x}'$ is the vector pointing to the scatterer from the incoming wave. When dealing with a finite range potential, we are interested in studying the system at a point far outside the nonvanishing area of the potential, i.e. we don't want to put a detector inside the scatter potential. Generally this means that we can set
\begin{equation}
    \abs{\vec{x}} \gg \abs{\vec{x}'}.
\end{equation}

We introduce $r=\abs{\vec{x}}$, $r' = \abs{\vec{x}'}$, and $\alpha = \angle (\vec{x}, \vec{x}')$. For $r\gg r'$, we write
\begin{alphaAlign}
    \abs{\vec{x}-\vec{x}'} &= \sqrt{r^2 - 2 r r' \cos\alpha + r'^2}\\
    &= r\sqrt{1 - \frac{2r'}{r} \cos\alpha + \frac{r'^2}{r^2} }\\
    &\approx r - \hv{r} \cdot \vec{x}',
\end{alphaAlign}
where
\begin{equation}
    \hv{r} = \frac{\vec{x}}{\abs{\vec{x}}}, \quad \text{and} \quad \vec{k}' = k \hv{r}.
\end{equation}
With this we write
\begin{equation}
    e^{\pm i k \abs{\vec{x-\vec{x}'}}} \approx e^{\pm i k r} e^{\mp i \vec{k}' \cdot \vec{x}'},
\end{equation}
and for large $r$ it is also valid to replave $1/\abs{\vec{x} - \vec{x}'}$ by $1/r$.

Choosing the initial state as an eigenstate to the free particle Hamiltonian $H_0$, such that $\k{i} = \k{\vec{k}}$, we can write
\begin{alphaAlign}
    \bk{\vec{x}}{\psi^{(+)}} &\overset{r \text{ large}}{\rightarrow} \bk{\vec{x}}{\vec{k}} - \frac{1}{4\pi} \frac{2m}{\hbar^2} \frac{e^{ikr}}{r} \int d^3 x' e^{-i \vec{k}' \cdot \vec{x}'} V(\vec{x}') \bk{\vec{x}'}{\psi^{(+)}}\\
    &= \frac{1}{L^{3/2}} \ab [e^{i\vec{k}\cdot \vec{x}} + \frac{e^{ikr}}{r} f(\vec{k}', \vec{k})].
\end{alphaAlign}
Here we see clearly that the scattered wave includes both the incident plane wave, and the outgoing spherical wave with amplitude $f(\vec{k}', \vec{k})$ given by
\begin{alphaAlign}
    f(\vec{k}',\vec{k}) &= \frac{1}{4\pi} \frac{2m}{\hbar^2} L^3 \int d^3 x' \frac{e^{-i \vec{k}'\cdot\vec{x}'}}{L^{3/2}} V(\vec{x}') \bk{\vec{x}'}{\psi^{(+)}}\\
    &=- \frac{m L^3}{2\pi\hbar^2} \bkop{\vec{k}'}{V}{\psi^{(+)}}.
\end{alphaAlign}
Similarly, we can can show for $\k{\psi^{-}}$ that
\begin{equation}
    \bk{\vec{x}}{\psi^{(-)}} = \frac{1}{L^{3/2}} \ab [ e^{i\vec{k}\cdot \vec{x}} + \frac{e^{-ikr}}{r} f(\vec{k}', \vec{k})]
\end{equation}
and
\begin{equation}
    f(\vec{k}', \vec{k}) = -\frac{m L^3}{2 \pi \hbar^2} \bkop{-\vec{k}'}{V}{\psi^{(-)}}.
\end{equation}
We call $f(\vec{k}', \vec{k})$ the scattering amplitude, and we remark that the differential cross section can be written
\begin{equation}
    \frac{d\sigma}{d\Omega} = \abs{f(\vec{k}',\vec{k})}^2.
\end{equation}

\subsubsection{Wave Packet Description}
This description is not valid for a particle being scattered by a scattering center if we model the particle as a wave packet approaching a scattering center. This is hard, but we notice that if the diemsnions of the wave packet is much larger than the size of the scatterer, i.e. the range of $V$, the plane wave approach is valid.

\subsubsection{The Optical Theorem}
A useful and fundamental relationship in quantum mechanics is the optics theorem, which relates the imaginary part of the forward scattering amplitude $f(\vec{k},\vec{k})$ ($\theta = 0$), to the total cross section $\sigma_\text{tot} = \int d\Omega (d\sigma / d\Omega)$, and it reads
\begin{equation}
    \Im f(\vec{k},\vec{k}) = \frac{k \sigma_\text{tot}}{4\pi}.
\end{equation}
The proof of this theorem start with the Lippmann-Schwinger equation, with $\k{i} \to \k{\vec{k}}$
\begin{equation}
    \k{\psi^{(+)}} = \k{\vec{k}} + \frac{1}{E - H_0 + i \epsilon} V \k{\psi^{(+)}}
\end{equation}
and we can rewrite this to obtain
\begin{equation}
    \k{\vec{k}} =  \k{\psi^{(+)}} - \frac{1}{E - H_0 + i \epsilon} V \k{\psi^{(+)}}.
\end{equation}
We can use this and write
\begin{alphaAlign}
    \bkop{\vec{k}}{V}{\psi^{(+)}} &= \ab [ \b{\psi^{(+)}} - \b{\psi^{(+)}} V \frac{1}{E - H_0 - i\epsilon} ] V \k{\psi^{(+)}}\\
    &= \bkop{\psi^{(+)}}{V}{\psi^{(+)}} - \bkop{\psi^{(+)}}{V \frac{1}{E-H_0-i\epsilon} V}{\psi^{(+)}}.
\end{alphaAlign}
We want to take the imaginary part of both sides of this expression. The first term on the right is the expectation value of a Hermitian operator and is thus real. To calculate the imaginary part of the second term we need to use complex analysis and Cauchy's principle value as we get a singularity when $\epsilon \to 0$. Since $\epsilon > 0$, we use a contour integral with a small semicircle extending down to the lower half plane around the singularity. Let this semicircle have the radius $\delta$, and let the singularity be at $z_0 = x_0 + i\epsilon$. To show the trick we will use, let us consider a complex function $f(z)= x + iy$, and we can then write
\begin{equation}
    \int_{-\infty}^{\infty} dx \frac{f(x)}{x-x_0} = \mathscr{P} \int_{-\infty}^{\infty} dx \frac{f(x)}{x-x_0} + \int_c dz \frac{f(z)}{z-z_0},
\end{equation}
where $c$ denotes the semicircle contour, and the Cauchy principle vlaue is defined as
\begin{equation}
    \mathscr{P} \int_{-\infty}^{\infty} dx \frac{f(x)}{x-x_0} = \lim_{\delta \to 0} \ab\{ \int_{-\infty}^{x_0 - \delta} dx \frac{f(x)}{x-x_0}  + \int_{x_0 + \delta}^{\infty}dx \frac{f(x)}{x-x_0}\}.
\end{equation}
For the second term we can use calculus of residues to evaluate
\begin{equation}
    \int_c dz \frac{f(z)}{z-z_0} = i\pi f(x_0) \quad \text{as }\delta \to 0.
\end{equation}
We now have 
\begin{equation}
    \int_{-\infty}^{\infty} dx \frac{f(x)}{x-x_0} = \mathscr{P} \int_{-\infty}^{\infty} dx \frac{f(x)}{x-x_0} + i\pi f(x_0).
\end{equation}

Returning to the equation at hand, we start by writing
\begin{alphaAlign}
    \lim_{\epsilon \to 0} \ab [\frac{1}{E - H_0 - i\epsilon}] &= \lim_{\epsilon\to 0} \int_{-\infty}^{\infty} dE' \frac{\delta(E-E')}{E' - H_0 - i \epsilon}\\
    &= i\pi \delta(E-H_0),
\end{alphaAlign}
using what we showed earlier, and we have omitted the principle value since it is a real number. Recalling $T\k{\vec{k}} = V\k{\psi^{(+)}}$, we can write
\begin{equation}
    \Im \bkop{\vec{k}}{V}{\psi^{(+)}} = -\pi \bkop{\vec{k}}{T^\dagger \delta(E-H_0)T}{\vec{k}}.
\end{equation}
Taking the imaginary part of the forward scattering amplitude gives
\begin{alphaAlign}
    \Im f(\vec{k},\vec{k}) &= -\frac{mL^3}{2\pi \hbar^2} \Im \bkop{\vec{k}}{V}{\psi^{(+)}}\\
    &= \frac{m L^3}{2 \hbar^2} \bkop{\vec{k}}{T^\dagger \delta(E-H_0)T}{\vec{k}}\\
    &= \frac{m L^3}{2 \hbar^2} \sum_{\vec{k}'} \bkop{\vec{k}}{T^\dagger \delta(E-H_0)}{\vec{k}'} \bkop{\vec{k}'}{T}{\vec{k}}\\
    &= \frac{m L^3}{2\hbar^2} \sum_{\vec{k}'} \abs{\bkop{\vec{k}'}{T}{\vec{k}}}^2 \delta_{E, \frac{\hbar^2 \vec{k}'^2}{2m}}, 
\end{alphaAlign}
where $E=\hbar^2k^2/2m$. To simplify, we start with
\begin{equation}
    \bkop{\vec{k}'}{T}{\vec{k}} = \bkop{\vec{k}'}{V}{\psi^{(+)}} = -\frac{2\pi\hbar^2}{m L^3} f(\vec{k}', \vec{k}),
\end{equation}
and using 
\begin{equation}
    \sum_{\vec{k}} \to \frac{L^3}{(2\pi)^3} \int d^3 k,
\end{equation}
leading to
\begin{alphaAlign}
    \Im f(\vec{k}, \vec{k}) &= \frac{m L^3}{2 \hbar^2} \ab (\frac{2\pi \hbar^2}{m L^3})^2 \sum_{\vec{k}'} \abs{f(\vec{k}', \vec{k})}^2 \delta_{E, \frac{\hbar^2 \vec{k}'^2}{2m}} \\
    &\to \frac{2\pi^2 \hbar^2}{m L^3} \frac{L^3}{(2\pi)^3} \int d^3 k' \abs{f(\vec{k}', \vec{k})}^2 \delta\ab(E -  \frac{\hbar^2 \vec{k}'^2}{2m})\\
    &= \frac{\hbar^2}{4\pi m} \int d^3 k' \frac{d\sigma}{d\Omega_{k'}}\delta\ab(E -  \frac{\hbar^2 \vec{k}'^2}{2m}).
\end{alphaAlign}
We have the volume element $d^3k' = k'^2 dk' d\Omega_{k'}$, so we have
\begin{alphaAlign}
    \Im f(\vec{k},\vec{k}) &= \frac{\hbar^2}{4\pi m} \int d\Omega_{k'} \frac{d\sigma}{d\Omega_{k'}} \int d k' k'^2 \delta\ab(E -  \frac{\hbar^2 \vec{k}'^2}{2m})\\
    &= \frac{\hbar^2}{4\pi m} \frac{m}{k \hbar^2} \int d\sigma \int dk' k'^2 \delta(k' - k)\\
    &= \frac{k}{4\pi} \sigma_\text{tot},
\end{alphaAlign}
where we have used the identity for the delta function
\begin{equation}
    \delta(g(x)) = \frac{\delta(x-x_0)}{\abs{\partial_x g(x_0)}}.
\end{equation}
